## Endpoints

- /api/v1/hotels
    - '/' GET getAllHotels
    - '/' POST createHotel
    - '/:id' GET getHotelById
    - '/:id' PATCH updateHotel
    - '/:id' DELETE deleteHotel

- /api/v1/users
    - '/signup' signup
    - '/login' login

- /api/v1/reviews
    -  '/' GET getAllReviews
    -  '/' POST createReview
